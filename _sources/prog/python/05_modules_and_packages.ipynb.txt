                "\n",
                "The core idea is simple:\n",
                "1. Define functions in one file (for example, `my_math.py`).\n",
                "2. Import and use them in another file (for example, `main.py`).\n",
                "\n",
                "This is the most common module workflow in Python projects."
            ]
        },
        {
            "cell_type": "code",

            "source": [
                "# Step 1: Create module file: my_math.py\n",
                "from pathlib import Path\n",
                "\n",
                "Path(\"my_math.py\").write_text(\n",
                "    \"def add(a, b):\\n\"\n",
                "    \"    return a + b\\n\\n\"\n",
                "    \"def multiply(a, b):\\n\"\n",
                "    \"    return a * b\\n\"\n",
                ")\n",
                "\n",
                "# Step 2: Create caller file: main.py\n",
                "Path(\"main.py\").write_text(\n",
                "    \"from my_math import add, multiply\\n\\n\"\n",
                "    \"print(add(2, 3))\\n\"\n",
                "    \"print(multiply(4, 5))\\n\"\n",
                ")\n",
                "\n",
                "# Step 3: Run main.py to use functions from my_math.py\n",
                "exec(Path(\"main.py\").read_text())"
